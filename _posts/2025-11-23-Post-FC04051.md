import turtle
import random



def tree(l,a,s,d,lc):
    if l>30:
        turtle.pensize(s)
        turtle.forward(l)
        turtle.right(a)
        tree(l*0.7, a + random.uniform(-5,5), s-2, d, lc)
        turtle.left(2*a)
        tree(l*0.7, a + random.uniform(-5,5), s-2, d, lc)
        turtle.right(a)
        turtle.backward(l)
    else:
        turtle.forward(l)
        color=random.choice(lc)
        turtle.dot(d,color)
        turtle.backward(l)
